🗄️ Firestore Security Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── Helpers ─────────────────────────────────────────────────
    function isSignedIn()  { return request.auth != null; }
    function uid()         { return request.auth.uid; }
    function isProf()      {
      return isSignedIn() &&
        get(/databases/$(database)/documents/users/$(uid())).data.role == 'prof';
    }
    function isSelf(u)     { return isSignedIn() && uid() == u; }

    // ── users/ ───────────────────────────────────────────────────
    // Users can read/write their own profile. Profs can read all.
    match /users/{userId} {
      allow read:   if isSelf(userId) || isProf();
      allow create: if isSelf(userId);
      allow update: if isSelf(userId) || isProf();
    }

    // ── questions_public/ ────────────────────────────────────────
    // All signed-in users can read. Only profs can write.
    match /questions_public/{qId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── questions_private/ ───────────────────────────────────────
    // Hidden test cases — professors only.
    match /questions_private/{qId} {
      allow read, write: if isProf();
    }

    // ── exams/ ───────────────────────────────────────────────────
    // Students read active exams. Profs have full access.
    match /exams/{examId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── submissions/ ─────────────────────────────────────────────
    // Students create their own. Profs read all.
    match /submissions/{subId} {
      allow read:   if isProf();
      allow create: if isSignedIn();
      allow update: if isProf();
    }

    // ── violations/ ──────────────────────────────────────────────
    // Students write their own violations. Profs read all.
    match /violations/{vioId} {
      allow read:   if isProf();
      allow create: if isSignedIn();
    }

    // ── drafts/ ──────────────────────────────────────────────────
    // Students read/write their own drafts only.
    match /drafts/{draftId} {
      allow read, write: if isSignedIn() &&
        resource.data.userId == uid();
      allow create: if isSignedIn();
    }

    // ── semesters/ ───────────────────────────────────────────────
    match /semesters/{semId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── departments/ ─────────────────────────────────────────────
    match /departments/{deptId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── halls/ ───────────────────────────────────────────────────
    match /halls/{hallId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── hall_allocations/ ────────────────────────────────────────
    match /hall_allocations/{allocId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── students/ ────────────────────────────────────────────────
    // Students read their own record. Profs have full access.
    match /students/{rollNo} {
      allow read:   if isProf() ||
        (isSignedIn() && resource.data.email == request.auth.token.email);
      allow create, update: if isProf();
    }

    // ── attendance/ ──────────────────────────────────────────────
    match /attendance/{examId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── monitor/ (live student heartbeats) ───────────────────────
    match /monitor/{userId} {
      allow read:   if isProf();
      allow write:  if isSelf(userId);
    }
  }
}
