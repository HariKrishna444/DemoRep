Firestore Security Rules
Rules are managed directly in your Firebase Console under Firestore → Rules. No rules are embedded in this app.
Open Firebase Console ↗
✅ Firestore rules are handled server-side in your Firebase project.
✅ This app does not store or copy security rules locally.
⚡ Navigate to Firebase Console → your project → Firestore Database → Rules to view or edit. 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── Helpers ─────────────────────────────────────────────────
    function isSignedIn()  { return request.auth != null; }
    function uid()         { return request.auth.uid; }
    // Safe isProf: uses exists() before get() so a missing /users/{uid}
    // document never causes a permission-denied cascade on other collections.
    function isProf() {
      return isSignedIn() &&
        exists(/databases/$(database)/documents/users/$(uid())) &&
        get(/databases/$(database)/documents/users/$(uid())).data.role == 'prof';
    }
    function isSelf(u)     { return isSignedIn() && uid() == u; }

    // ── users/ ───────────────────────────────────────────────────
    // Users can read/write their own profile. Profs can read all.
    match /users/{userId} {
      allow read:   if isSelf(userId) || isProf();
      allow create: if isSelf(userId);
      allow update: if isSelf(userId) || isProf();
    }

    // ── questions_public/ ────────────────────────────────────────
    // All signed-in users can read. Only profs can write.
    match /questions_public/{qId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── questions_private/ ───────────────────────────────────────
    // Hidden test cases — professors only.
    match /questions_private/{qId} {
      allow read, write: if isProf();
    }

    // ── exams/ ───────────────────────────────────────────────────
    // Students read active exams. Profs have full access.
    match /exams/{examId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── submissions/ ─────────────────────────────────────────────
    // Students create their own. Profs read all.
    match /submissions/{subId} {
      allow read:   if isProf();
      allow create: if isSignedIn();
      allow update: if isProf();
    }

    // ── violations/ ──────────────────────────────────────────────
    // Students write their own violations. Profs read all.
    match /violations/{vioId} {
      allow read:   if isProf();
      allow create: if isSignedIn();
    }

    // ── drafts/ ──────────────────────────────────────────────────
    // Students read/write their own drafts only.
    match /drafts/{draftId} {
      allow read, write: if isSignedIn() &&
        resource.data.userId == uid();
      allow create: if isSignedIn();
    }

    // ── semesters/ ───────────────────────────────────────────────
    match /semesters/{semId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── departments/ ─────────────────────────────────────────────
    match /departments/{deptId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── halls/ ───────────────────────────────────────────────────
    match /halls/{hallId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── hall_allocations/ ────────────────────────────────────────
    match /hall_allocations/{allocId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── students/ ────────────────────────────────────────────────
    // Students read their own record. Profs have full access.
    match /students/{rollNo} {
      allow read:   if isProf() ||
        (isSignedIn() && resource.data.email == request.auth.token.email);
      allow create, update: if isProf();
    }

    // ── attendance/ ──────────────────────────────────────────────
    match /attendance/{examId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── monitor/ (live student heartbeats) ───────────────────────
    match /monitor/{userId} {
      allow read:   if isProf();
      allow write:  if isSelf(userId);
    }

    // ── courses/ — course catalog ─────────────────────────────────
    // Courses are public (catalog). Quizzes require sign-in to prevent
    // answer scraping by anonymous users.
    match /courses/{courseId} {
      allow read:  if true;        // public catalog
      allow write: if isProf();
    }
    match /courses/{courseId}/modules/{moduleId} {
      allow read:  if true;
      allow write: if isProf();
    }
    match /courses/{courseId}/modules/{moduleId}/lessons/{lessonId} {
      allow read:  if true;
      allow write: if isProf();
    }
    match /courses/{courseId}/quizzes/{quizId} {
      // SECURITY FIX: this doc now only ever contains question text +
      // options (see saveQuiz() in the client) — correctIndex was removed
      // and moved to quiz_answers/ below. Any signed-in student reading
      // this doc directly (e.g. via devtools) no longer gets the answer key.
      allow read:  if isSignedIn();
      allow write: if isProf();
    }
    match /quiz_answers/{docId} {
      // Answer key — never readable by the client, signed in or not.
      // Only the verifyQuiz Cloud Function reads this, via the Admin SDK
      // (which bypasses these rules entirely).
      allow read:  if false;
      allow write: if isProf();
    }
    match /rateLimits/{uid} {
      // Per-user sliding-window counters used by the runCodeSandbox Cloud
      // Function. Written only by the Function (Admin SDK bypasses rules);
      // no client read/write needed at all.
      allow read, write: if false;
    }

    // ── enrollments/ — student progress ──────────────────────────
    match /enrollments/{docId} {
      allow read:  if isSignedIn() && (isProf() || resource.data.userId == uid());
      allow create: if isSignedIn() && request.resource.data.userId == uid();
      allow update: if isSignedIn() && resource.data.userId == uid();
    }

    // ── certificates/ — issued certificates ───────────────────────
    match /certificates/{docId} {
      allow read:  if true;  // public — needed for verify page
      allow create: if isSignedIn() && request.resource.data.userId == uid();
      allow update: if isProf();
    }

    // ── practice_questions/ ────────────────────────────────────────
    // Untimed self-practice bank. Anyone signed in can read (students
    // need this to browse/solve practice problems); testcases include
    // `expected` output here since grading happens client-side, not
    // via a secure Cloud Function like exam questions. Only profs write.
    match /practice_questions/{pqId} {
      allow read:  if isSignedIn();
      allow write: if isProf();
    }

    // ── module_progress/ — per-module completion tracking ─────────
    match /module_progress/{docId} {
      allow read:  if isProf() || (isSignedIn() && resource.data.userId == uid());
      allow write: if isSignedIn() && request.resource.data.userId == uid();
    }
